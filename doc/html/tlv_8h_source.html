101 static inline gen_tlv_callback_t get_gen_cb_for_type(enum types type)
102 {
103  switch(type)
104  {
105  case IP_ADDR:
106  return gen_tlv_uint32;
107  case PACKING:
108  return gen_tlv_char;
109  case MAXCID:
110  return gen_tlv_uint32;
111  case UNID:
112  return gen_tlv_char;
113  case WINDOWSIZE:
114  return gen_tlv_uint32;
115  case REFRESH:
116  return gen_tlv_uint32;
117  case KEEPALIVE:
118  return gen_tlv_uint32;
119  case ROHC_COMPAT:
120  return gen_tlv_char;
121  case CPACKING:
122  return gen_tlv_char;
123  case CPROTO_VERSION:
124  return gen_tlv_char;
125  default:
126  return NULL;
127  }
