36 /**
37  * @brief This is a version of ip_compute_csum() optimized for IP headers,
38  * which always checksum on 4 octet boundaries.
39  *
40  * @author Jorge Cwik <jorge@laser.satlink.net>, adapted for linux by
41  * Arnt Gulbrandsen.
42  *
43  * @param iph The IPv4 header
44  * @param ihl The length of the IPv4 header
45  * @return The IPv4 checksum
46  */
47 static inline unsigned short ip_fast_csum(unsigned char *iph, unsigned int ihl)
48 {
49  unsigned int sum;
50 
51  __asm__ __volatile__ (
52  " \n\
53  movl (%1), %0 \n\
54  subl $4, %2 \n\
55  jbe 2f \n\
56  addl 4(%1), %0 \n\
57  adcl 8(%1), %0 \n\
58  adcl 12(%1), %0 \n\
59 1: adcl 16(%1), %0 \n\
60  lea 4(%1), %1 \n\
61  decl %2 \n\
62  jne 1b \n\
63  adcl $0, %0 \n\
64  movl %0, %2 \n\
65  shrl $16, %0 \n\
66  addw %w2, %w0 \n\
67  adcl $0, %0 \n\
68  notl %0 \n\
69 2: \n\
70  "
71  /* Since the input registers which are loaded with iph and ipl
72  are modified, we must also specify them as outputs, or gcc
73  will assume they contain their original values. */
74  : "=r" (sum), "=r" (iph), "=r" (ihl)
75  : "1" (iph), "2" (ihl));
76 
77  return(sum);
78 }

83 static inline unsigned short from32to16(unsigned long x)
84 {
85  /* add up 16-bit and 16-bit for 16+c bit */
86  x = (x & 0xffff) + (x >> 16);
87  /* add up carry.. */
88  x = (x & 0xffff) + (x >> 16);
89  return x;
90 }

93 static unsigned int do_csum(const unsigned char *buff, int len)
94 {
95  int odd, count;
96  unsigned long result = 0;
97 
98  if(len <= 0)
99  {
100  goto out;
101  }
102  odd = 1 & (unsigned long) buff;
103  if(odd)
104  {
105 #ifdef __LITTLE_ENDIAN
106  result = *buff;
107 #else
108  result += (*buff << 8);
109 #endif
110  len--;
111  buff++;
112  }
113  count = len >> 1; /* nr of 16-bit words.. */
114  if(count)
115  {
116  if(2 & (unsigned long) buff)
117  {
118  result += *(unsigned short *) buff;
119  count--;
120  len -= 2;
121  buff += 2;
122  }
123  count >>= 1; /* nr of 32-bit words.. */
124  if(count)
125  {
126  unsigned long carry = 0;
127  do
128  {
129  unsigned long w = *(unsigned int *) buff;
130  count--;
131  buff += 4;
132  result += carry;
133  result += w;
134  carry = (w > result);
135  }
136  while(count);
137  result += carry;
138  result = (result & 0xffff) + (result >> 16);
139  }
140  if(len & 2)
141  {
142  result += *(unsigned short *) buff;
143  buff += 2;
144  }
145  }
146  if(len & 1)
147 #ifdef __LITTLE_ENDIAN
148  { result += *buff; }
149 #else
150  { result += (*buff << 8); }
151 #endif
152  result = from32to16(result);
153  if(odd)
154  {
155  result = ((result >> 8) & 0xff) | ((result & 0xff) << 8);
156  }
157 out:
158  return result;
159 }
